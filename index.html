// دالة عرض التقرير الشامل لجميع الفصول (معدلة مع إضافة زر تصدير Excel) async function renderAllClassesReport(allStudents, attendanceRecords, startDate, endDate) { const classMap = new Map(allClasses.map(c => [c.id, c.class_name])); const studentMap = new Map(allStudents.map(s => [s.id, { name: s.name, class_id: s.class_id }])); // تجميع البيانات حسب الفصل const classesData = {}; allClasses.forEach(cls => { const classStudents = allStudents.filter(s => s.class_id === cls.id); const classAttendance = attendanceRecords.filter(r => classStudents.some(s => s.id === r.student_id)); classesData[cls.id] = { className: cls.class_name, students: classStudents, attendance: classAttendance }; }); // إحصائيات سريعة لكل فصل const statsHTML = allClasses.map(cls => { const data = classesData[cls.id]; const presentCount = data.attendance.filter(r => r.mass_grade === 10 && r.communion_grade === 10 && r.class_grade === 10).length; return `

${cls.class_name}

المخدومين: ${data.students.length}
الحضور الكامل: ${presentCount}
`; }).join(''); // عرض التقرير في adminContent document.getElementById('adminContent').innerHTML = `
تقرير شامل لجميع الفصول
${statsHTML}
${allClasses.map(cls => { const data = classesData[cls.id]; if (data.attendance.length === 0) return ''; const summaryData = data.students.map(s => { const studentRecords = data.attendance.filter(r => r.student_id === s.id); const totalDays = studentRecords.length; const totalMass = studentRecords.reduce((sum, r) => sum + (r.mass_grade || 0), 0); const totalCommunion = studentRecords.reduce((sum, r) => sum + (r.communion_grade || 0), 0); const totalClass = studentRecords.reduce((sum, r) => sum + (r.class_grade || 0), 0); const totalBehavior = studentRecords.reduce((sum, r) => sum + (r.behavior_grade || 0), 0); const grandTotal = totalMass + totalCommunion + totalClass + totalBehavior; return { name: s.name, totalDays, totalMass, totalCommunion, totalClass, totalBehavior, grandTotal }; }).filter(s => s.totalDays > 0); return `

${cls.class_name}

${summaryData.map(s => ` `).join('')}
الاسم عدد أيام الحضور إجمالي القداس إجمالي التناول إجمالي الخدمة إجمالي السلوك المجموع الكلي
${s.name} ${s.totalDays} ${s.totalMass} ${s.totalCommunion} ${s.totalClass} ${s.totalBehavior} ${s.grandTotal}
`; }).join('')}
`; // ربط أحداث الأزرار document.getElementById('backToAdminMain').addEventListener('click', () => { loadAdminView(); }); document.getElementById('fetchAllRangeBtn').addEventListener('click', async () => { const newStart = document.getElementById('startDateAll').value; const newEnd = document.getElementById('endDateAll').value; const newAttendance = await fetchAttendanceForAllClasses(newStart, newEnd); const allStudents = await fetchAllStudents(); renderAllClassesReport(allStudents, newAttendance, newStart, newEnd); }); // زر تصدير Excel الشامل document.getElementById('exportAllExcelBtn').addEventListener('click', () => { // تجهيز بيانات التصدير const exportData = []; allClasses.forEach(cls => { const data = classesData[cls.id]; if (data.attendance.length === 0) return; const summaryData = data.students.map(s => { const studentRecords = data.attendance.filter(r => r.student_id === s.id); const totalDays = studentRecords.length; const totalMass = studentRecords.reduce((sum, r) => sum + (r.mass_grade || 0), 0); const totalCommunion = studentRecords.reduce((sum, r) => sum + (r.communion_grade || 0), 0); const totalClass = studentRecords.reduce((sum, r) => sum + (r.class_grade || 0), 0); const totalBehavior = studentRecords.reduce((sum, r) => sum + (r.behavior_grade || 0), 0); const grandTotal = totalMass + totalCommunion + totalClass + totalBehavior; return { class_name: cls.class_name, student_name: s.name, total_days: totalDays, total_mass: totalMass, total_communion: totalCommunion, total_class: totalClass, total_behavior: totalBehavior, grand_total: grandTotal }; }).filter(s => s.total_days > 0); exportData.push(...summaryData); }); // إنشاء ورقة Excel const wsData = [['الفصل', 'الاسم', 'عدد أيام الحضور', 'إجمالي القداس', 'إجمالي التناول', 'إجمالي الخدمة', 'إجمالي السلوك', 'المجموع الكلي']]; exportData.forEach(row => { wsData.push([ row.class_name, row.student_name, row.total_days, row.total_mass, row.total_communion, row.total_class, row.total_behavior, row.grand_total ]); }); const wb = XLSX.utils.book_new(); const ws = XLSX.utils.aoa_to_sheet(wsData); XLSX.utils.book_append_sheet(wb, ws, 'تقرير شامل'); XLSX.writeFile(wb, `تقرير_شامل_${new Date().toLocaleDateString()}.xlsx`); }); }
